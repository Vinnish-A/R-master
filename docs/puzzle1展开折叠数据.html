infect_ = Cattle$infect
cattle_ = Cattle$cattle

Cattle_expanded = copy(Cattle)[rep(id, cattle), ]
Cattle_expanded[, infect := unlist(map2(infect_, cattle_ - infect_, ~ rep(c(1, 0), c(.x, .y))))]
